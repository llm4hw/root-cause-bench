â€ƒâ€ƒâ€¢ inverter[i].in = lut_outs[iâ€“1] for i = 1, 2, â€¦, 10

    input wire enable,
    input wire ICE_CLK,
    output reg ICE_LED,
    output reg OUT
);

    wire [10:0] lut_outs;

    generate

                .in(lut_ins[i]),
                .out(lut_outs[i])

        end

    genvar j;
    generate
        for (j = 1; j < 11; j = j + 1) begin : chain
            assign lut_ins[j] = lut_outs[j-1];
        end
    endgenerate

    reg [15:0] counter = 0;

        if (enable) begin
            counter <= counter + 1;
            ICE_LED <= counter[0];
            OUT <= counter[0];
        end
    end

    input wire in,
    output reg out
);
    always @(*) #1 out = ~in;
endmodule
